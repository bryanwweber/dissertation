\newcommand{\deflisttick}[1]{
\gdef\listticks{}
\gdef\listlabels{}
\gdef\minorticks{}
\foreach \x[count=\xi] in {#1}{\global\let\maxitems\xi}
\foreach \x[count=\xi] in {#1}{
\pgfmathsetmacro{\xx}{1000/\x}
\pgfmathsetmacro{\m}{1000/(\x+10)}
\pgfmathsetmacro{\mm}{1000/(\x+20)}
\pgfmathsetmacro{\mmm}{1000/(\x+30)}
\pgfmathsetmacro{\mmmm}{1000/(\x+40)}
 \ifnum\xi=\maxitems
   \xdef\listticks{\listticks \xx}
   \xdef\listlabels{\listlabels \SI{\x}{\kelvin}}
   \xdef\minorticks{\minorticks \m,\mm,\mmm,\mmmm}
 \else
   \xdef\listticks{\listticks \xx,}
   \xdef\listlabels{\listlabels \SI{\x}{\kelvin},}
   \xdef\minorticks{\minorticks \m,\mm,\mmm,\mmmm,}
 \fi
}
}
